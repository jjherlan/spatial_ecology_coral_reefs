soil <- read.csv('soil_full.csv') %>%
  as_tibble() %>%
  mutate_at(vars(material, range, range_mean), factor) %>%
  group_by(material, range_mean) %>%
  dplyr::summarize(mean = mean(mbc), 
                   sd = sd(mbc), 
                   n = n(),
                   se = sd/sqrt(n)
  ) %>%
  mutate(se = sd / sqrt(n),
         lower.ci = mean - qt(1 - (0.05 / 2), n - 1) * se,
         upper.ci = mean + qt(1 - (0.05 / 2), n - 1) * se)
